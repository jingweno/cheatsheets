Git alias shortcuts in ZSH.

## gst
gst = git status

## gd
gd = git diff

## gdc
gdc = git diff --cached

## gl
gl = git pull

## gup
gup = git pull --rebase

## gp
gp = git push

## gd
gd = git diff

## gc
gc = git commit -v

## gc!
gc! = git commit -v --amend

## gca
gca = git commit -v -a

## gca!
gca! = git commit -v -a --amend

## gcmsg
gcmsg = git commit -m

## gco
gco = git checkout

## gr
gr = git remote

## grbi
grbi = git rebase -i

## gb
gb = git branch

## gba
gba = git branch -a

## gcount
gcount = git shortlog -sn

## gcp
gcp = git cherry-pick

## glg
glg = git log --stat --max-count=10

## glgg
glgg = git log --graph --max-count=10

## glo
glo = git log --oneline --decorate --color

## gss
gss = git status -s

## ga
ga = git add

## gm
gm = git merge

## grhh
grhh = git reset HEAD --hard

## gmt
gmt = git mergetool --no-prompt

## gg
gg = git gui citool

## gk
gk = gitk --all --branches
